void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: Text('Universal File Viewer')),
        body: Center(
          child: UniversalFileViewer(filePath: '/storage/emulated/0/Download/sample.pdf'),
        ),
      ),
    );
  }
